// Função para criar os elementos do perfil de um membro
function createTeamMemberElement(member) {
    const container = document.createElement("div");

    const img = document.createElement("img");
    img.src = member.imgSrc;
    img.alt = "profile";

    const name = document.createElement("span");
    name.className = "profile-bold";
    name.textContent = member.name;

    const role = document.createElement("span");
    role.className = "profile";
    role.textContent = member.role;

    const resume = document.createElement("span");
    resume.className = "align resume align";
    resume.textContent = member.resume;

    const links = document.createElement("div");
    links.className = "links";

    const githubLink = document.createElement("a");
    const githubImg = document.createElement("img");
    githubImg.src = "/static/media/github.c91972a5cb4ff66b22c0ecb084c7cc74.svg";
    githubImg.alt = "github";
    githubLink.appendChild(githubImg);

    const linkedinLink = document.createElement("a");
    linkedinLink.href = member.linkedin;
    linkedinLink.style.margin = "0px 15px";
    const linkedinImg = document.createElement("img");
    linkedinImg.src = "/static/media/linkedin.dfb5c1d4fb3dbb442c54088985e0f221.svg";
    linkedinImg.alt = "linkedin";
    linkedinLink.appendChild(linkedinImg);

    const lattesLink = document.createElement("a");
    const lattesImg = document.createElement("img");
    lattesImg.src = "/static/media/lattes.6e86fe7ea37433b7fb688ba1e606471c.svg";
    lattesImg.alt = "lattes";
    lattesImg.style.width = "38px";
    lattesLink.appendChild(lattesImg);

    links.appendChild(githubLink);
    links.appendChild(linkedinLink);
    links.appendChild(lattesLink);

    container.appendChild(img);
    container.appendChild(name);
    container.appendChild(role);
    container.appendChild(resume);
    container.appendChild(links);

    return container;
}

// Função para adicionar os membros da equipe ao DOM
function addTeamToPage() {
    const teamSection = document.getElementById("our-team");

    teamMembers.forEach((member) => {
        const teamElement = createTeamMemberElement(member);
        teamSection.appendChild(teamElement);
    });
}

// Adiciona a equipe ao carregar a página
document.addEventListener("DOMContentLoaded", addTeamToPage);
